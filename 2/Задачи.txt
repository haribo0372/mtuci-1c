Функция ВалидФИО(Стр)
	перем Результат, РегулярноеВыр;
	
	//РегулярноеВыр = "^[А-ЯЁа-яё]+\s[А-ЯЁ]\.\s[А-ЯЁ]\.$";

	//Если Флаг = Истина Тогда
	//	Возврат "";
	//КонецЕсли;
	
	РазделСтр = СтрРазделить(Стр, " ", Ложь);
	Фамилия = РазделСтр[0];
	ИО = РазделСтр[1];
	ДлинаФам = СтрДлина(Фамилия);
	ДлинаИО = СтрДлина(ИО);
	Результат = ВРег(Лев(Фамилия, 1)) + НРег(Прав(Фамилия, ДлинаФам - 1)) + " " + Врег(ИО);
	
	Возврат Результат;
КонецФункции

Функция УдалитьЗнакиПрепинания(Строка)
	перем Результат, ЗнакиПреп;
	ЗнакиПреп = ".,;:!?-—()[]{}''«»";
	
	Результат = Строка;
	Для ИТ=1 по СтрДлина(ЗнакиПреп) Цикл
		Результат =  СтрЗаменить(Результат, Сред(ЗнакиПреп, ИТ, 1), "");
	КонецЦикла;
	
	Возврат Результат;
КонецФункции

функция ИнициализацияМассива(Знач1, Знач2)
	ВремМассив = Новый Массив(2);
	ВремМассив[0] = Знач1;
	ВремМассив[1] = Знач2;
	Возврат ВремМассив;
